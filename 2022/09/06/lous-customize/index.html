#include<iostream>
#include<vector>
#include<cassert>
#include<algorithm>
#include<unordered_map>

using namespace std;

class Solution {
public:
    vector<int> twoSum(const vector<int>& nums, int target){
        // assume that each input would have exactly one solution
        unordered_map<int, int> num_idx_dict;  // value -> index
        vector<int> res;
        for (int idx=0; idx<nums.size(); ++idx){
            auto find_iter = num_idx_dict.find(target-nums[idx]);
            if (find_iter == num_idx_dict.end()){
                // NOTE: emplaceçš„æ¥å£ç±»ä¼¼pair
                // num_idx_dict.emplace(nums[idx], idx);
                // NOTE: ä¹Ÿå¯ä»¥ç›´æ¥ç”¨ä¸‹æ ‡å¢åŠ å…ƒç´ 
                num_idx_dict[nums[idx]] = idx;
            }else{
                // NOTE: å¦‚æœè®¿é—®mapçš„è¿­ä»£å™¨ï¼Œæ¥å£ç±»ä¼¼pair
                // res.push_back((*find_iter).second);
                // NOTE: å¦‚æœæ˜¯é€šè¿‡ä¸‹æ ‡è®¿é—®ï¼Œè¿”å›å€¼
                res.push_back(num_idx_dict[target-nums[idx]]);
                res.push_back(idx);
                break;
            }
        }
        return res;
    }
};

int main(){
    Solution s;
    vector<int> res = s.twoSum(vector<int>{2,7,11,15}, 9);
    vector<int> ans{0,1};
    assert(res == ans);
}
